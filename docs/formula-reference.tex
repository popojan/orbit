\section*{Pell Equations \& Square Root Rationalization}

\subsection*{Pell Equation}
For non-square integer $d$:
\begin{equation}
x^2 - d \cdot y^2 = 1
\end{equation}

Fundamental solution $(x_0, y_0)$ provides best rational approximation:
\begin{equation}
\frac{x_0}{y_0} \approx \sqrt{d}
\end{equation}

\subsection*{Wildberger's Pell Solver}
Iterative algorithm (Fibonacci-like):
\begin{align}
a_0 &= 1,\quad b_0 = 0,\quad c_0 = -d\\
t &= a + 2b + c\\
\text{If } t > 0&: \quad a \leftarrow t,\; b \leftarrow b + c\\
\text{If } t \leq 0&: \quad b \leftarrow b + a,\; c \leftarrow t
\end{align}

\subsection*{Chebyshev-Pell Refinement}

\textbf{Base approximation:}
\begin{equation}
b = \frac{x-1}{y} \quad \text{where } (x,y) \text{ is Pell solution}
\end{equation}

\textbf{Fast refinement (sqrttrf):}
\begin{multline}
\text{sqrttrf}(d, n, m) = \frac{n^2 + d}{2n} + \frac{n^2 - d}{2n} \cdot\\
\frac{U_{m-1}\left(\sqrt{\frac{d}{-(n^2-d)}}\right)}{U_{m+1}\left(\sqrt{\frac{d}{-(n^2-d)}}\right)}
\end{multline}

where $U_k$ is Chebyshev polynomial of second kind.

\textbf{Symmetric step:}
\begin{equation}
\text{sym}(d, n, m) = \frac{d}{2x} + \frac{x}{2}, \quad x = \text{sqrttrf}(d, n, m)
\end{equation}

\textbf{Nested iteration:}
\begin{equation}
\text{nestqrt}(d, n_0, \{m_1, m_2\}) = \underbrace{\text{sym}(\cdots\text{sym}(d, n_0, m_1)\cdots, m_1)}_{m_2 \text{ times}}
\end{equation}

\subsection*{m1=1 Optimization}

When $m_1 = 1$, sqrttrf simplifies dramatically:
\begin{equation}
\text{sqrttrf}(d, n, 1) = \frac{n^2 + d}{2n}
\end{equation}

Convergence: $\sim$6000$\times$ precision per iteration!

\subsection*{Error Metrics}

\textbf{Quadratic error:}
\begin{equation}
E_{\text{quad}} = \log_{10}|d - \text{approx}^2|
\end{equation}

\textbf{Sqrt precision} (decimal places):
\begin{equation}
P_{\text{sqrt}} = -\frac{E_{\text{quad}}}{2}
\end{equation}

\section*{Primorial Computation}

\subsection*{Primorial Definition}
\begin{equation}
p\# = \prod_{\text{primes } q \leq p} q
\end{equation}

\subsection*{Rational Sum Formula}
\begin{equation}
p\# = \left\lfloor \frac{1}{\sum_{k=1}^{p} \frac{1}{k!}} \right\rfloor
\end{equation}

\subsection*{Recursive Sieve Formulation}
\begin{align}
S_2(n) &= 2\left\lfloor \frac{n}{2} \right\rfloor\\
S_p(n) &= p \cdot S_{p'}(n') \quad \text{where } p' = \text{prev prime}\\
n' &= \left\lfloor \frac{n}{p} \right\rfloor - S_{p'}\left(\left\lfloor \frac{n}{p^2} \right\rfloor\right)
\end{align}

Primorial:
\begin{equation}
p\# = S_p(p)
\end{equation}

\section*{Semiprime Factorization}

For semiprime $n = p \cdot q$ where $p < q$:

\subsection*{Pochhammer Sum}
\begin{equation}
\text{ForFactiMod}(n) = \text{FractionalPart}\left(\sum_{k=1}^{n-1} \frac{(n-1)!}{(k-1)! \cdot k \cdot n}\right)
\end{equation}

\subsection*{Factor Extraction}
\begin{equation}
\text{ForFactiMod}(n) = \frac{p-1}{p}
\end{equation}

Therefore:
\begin{equation}
p = \frac{1}{1 - \text{ForFactiMod}(n)}
\end{equation}

\section*{Modular Factorials}

\subsection*{Half-Factorial Identity}

For prime $p$:
\begin{equation}
\left(\frac{p-1}{2}\right)! \equiv \begin{cases}
\pm 1 \pmod{p} & \text{if } p \equiv 3 \pmod{4}\\
\pm i \pmod{p} & \text{if } p \equiv 1 \pmod{4}
\end{cases}
\end{equation}

where $i^2 \equiv -1 \pmod{p}$.

\subsection*{Efficient Computation}

\textbf{Find $\sqrt{-1} \bmod p$:}
\begin{equation}
i \equiv 2^{(p-1)/4} \pmod{p}
\end{equation}

\textbf{Compute $n! \bmod p$:}
\begin{align}
n! &\equiv h \cdot \prod_{k=h+1}^{n} k \pmod{p}\\
\text{where } h &= \left(\frac{p-1}{2}\right)! \bmod p
\end{align}

\section*{Prime DAG \& Gap Theorem}

\subsection*{Greedy Prime Decomposition}

For integer $n$:
\begin{equation}
\text{PrimeRepSparse}(n) = \{p_1, p_2, \ldots, p_k, r\}
\end{equation}
where each $p_i = \pi(\text{remainder})$ (largest prime $\leq$ remainder).

\subsection*{Prime Orbit}

Recursive transitive closure:
\begin{equation}
\text{Orbit}(n) = \bigcup_{p \in \text{Decomp}(n)} \{p\} \cup \text{Orbit}(\pi(p))
\end{equation}

\subsection*{Gap Theorem}

For prime $p$ with gap $g = p_{+} - p$ to next prime:

\textbf{Theorem:} Exactly $g$ integers have $p$ in their orbit:
\begin{equation}
|\{n \in [\pi(p), \pi(p) + g) : p \in \text{Orbit}(n)\}| = g
\end{equation}

\section*{Chebyshev Visualization}

\subsection*{Curve Family}

\begin{equation}
f_k(x) = T_{k+1}(x) - x \cdot T_k(x)
\end{equation}

where $T_k$ is Chebyshev polynomial (first kind).

\subsection*{Properties}

\textbf{Unit norm:}
\begin{equation}
\int_{\theta=0}^{2\pi} |f_k(\cos\theta)| \, d\theta = 1
\end{equation}

\textbf{Regular polygon contact:} Curve touches unit circle at vertices of regular $(k+1)$-gon.

\textbf{Rotation angle:}
\begin{equation}
\theta_i = -\frac{\pi}{2k} + \frac{2\pi}{k+1} \cdot i
\end{equation}

\subsection*{Complex Number Formula}

\begin{multline}
f_k(x) = \frac{(-1)^k \cdot i\sqrt{1-x^2}}{2} \cdot\\
\left[(1-2x^2 - 2ix\sqrt{1-x^2})^k - (1-2x^2 + 2ix\sqrt{1-x^2})^k\right]
\end{multline}

\section*{Egyptian Horcruxes}

\subsection*{Multiple Representations}

A rational $q$ can have multiple Egyptian fraction decompositions:
\begin{equation}
q = \frac{1}{a_1} + \frac{1}{a_2} + \cdots + \frac{1}{a_n}
\end{equation}

Each complete set is a "Horcrux" - any one can resurrect the original.

\subsection*{Example: 23/29}

Generates $\geq 51$ different Horcrux sets, including:
\begin{align}
\frac{23}{29} &= \frac{1}{2} + \frac{1}{5} + \frac{1}{12} + \frac{1}{120} + \frac{1}{696}\\
\frac{23}{29} &= \frac{1}{2} + \frac{1}{6} + \frac{1}{9} + \frac{1}{86} + \cdots
\end{align}

\subsection*{Steganographic Property}

\textbf{Redundancy:} Multiple independent representations.

\textbf{Fragility:} Missing one fraction from a set $\Rightarrow$ cannot recover.

\textbf{Security:} Must destroy ALL sets to prevent recovery.
